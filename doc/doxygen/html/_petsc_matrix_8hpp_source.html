312 inline void PetscMatrix::add(const double a_in, // double constant
313  SparseMatrix &X_in // sparse matrix
314  ) {
315  assert(this->initialized());
316 
317  // crash due to incompatible sparsity structure...
318  assert(this->m() == X_in.m());
319  assert(this->n() == X_in.n());
320 
321  PetscScalar a = static_cast<PetscScalar>(a_in);
322  PetscMatrix* X = dynamic_cast<PetscMatrix*>(&X_in);
323 
324  // the matrix from which we copy the values has to be assembled/closed
325  // X->close ();
326  assert(X != NULL);
327  assert(X->closed());
328  semiparallel_only();
329  int ierr=0;
330 
331 #if PETSC_VERSION_LESS_THAN(2,3,0) // 2.2.x & earlier style
332  ierr = MatAXPY(&a, X->_mat, _mat, SAME_NONZERO_PATTERN);
333  CHKERRABORT(MPI_COMM_WORLD,ierr);
334 #else // 2.3.x & newer
335  ierr = MatAXPY(_mat, a, X->_mat, DIFFERENT_NONZERO_PATTERN);
336  CHKERRABORT(MPI_COMM_WORLD,ierr);
337 #endif
338 }

341 inline double PetscMatrix::operator()(const int i, // index i
342  const int j // index j
343  ) const {
344 
345  // do no use this function !!!!!!!!
346  assert(this->initialized());
347 #if PETSC_VERSION_LESS_THAN(2,2,1) // PETSc 2.2.0 & older
348  PetscScalar *petsc_row;
349  int* petsc_cols;
350 #else // PETSc 2.2.1 & newer
351  const PetscScalar *petsc_row;
352  const PetscInt *petsc_cols;
353 #endif
354 
355  // If the entry is not in the sparse matrix, it is 0.
356  double value=0.;
357  int ierr=0, ncols=0, i_val=static_cast<int>(i), j_val=static_cast<int>(j);
358 
359  assert(this->closed());// the matrix needs to be closed for this to work
360  ierr = MatGetRow(_mat, i_val, &ncols, &petsc_cols, &petsc_row);
361  CHKERRABORT(MPI_COMM_WORLD,ierr);
362  // Perform a binary search to find the contiguous index in
363  // petsc_cols (resp. petsc_row) corresponding to global index j_val
364  std::pair<const int*, const int*> p =
365  std::equal_range(&petsc_cols[0], &petsc_cols[0] + ncols, j_val);
366  // Found an entry for j_val
367  if (p.first != p.second) {
368  // The entry in the contiguous row corresponding to the j_val column of interest
369  const int j = std::distance(const_cast<int*>(&petsc_cols[0]),const_cast<int*>(p.first));
370 
371  assert(j < ncols);
372  assert(petsc_cols[j] == j_val);
373  value = static_cast<double>(petsc_row[j]);
374  }
375  ierr = MatRestoreRow(_mat, i_val,
376  &ncols, &petsc_cols, &petsc_row);
377  CHKERRABORT(MPI_COMM_WORLD,ierr);
378  return value;
379 }
