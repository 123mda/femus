217 inline
218 T_sys & MultiLevelProblem::add_system (const std::string& name,const MgSmoother & smoother_type )
219 {
220  T_sys* ptr = NULL;
221 
222  if (!_systems.count(name))
223  {
224  ptr = new T_sys(*this, name, this->n_systems(),smoother_type);
225  _systems.insert (std::make_pair(name, ptr));
226  }
227  else
228  {
229  // We now allow redundant add_system calls, to make it
230  // easier to load data from files for user-derived system
231  // subclasses
232  std::cerr << "ERROR: There was already a system"
233  << " named " << name
234  << std::endl;
235 
236  ptr = &(this->get_system<T_sys>(name));
237  }
238 
239  // Return a dynamically casted reference to the newly added System.
240  return *ptr;
241 }
